# EEG_AI_AR 月球幸存者 - LLM 情绪干预游戏系统

## 📋 项目概述

### 游戏背景
**用户角色**：一名宇航员，在月盾计划失败后独自留在月球，成为「宇宙最后的人类」
**核心设定**：用户处于失忆状态，需要通过与 AI 助手的对话逐步恢复记忆
**游戏目标**：通过对话探索和记忆恢复，最终达到记忆恢复的临界值，触发游戏结局

### 目标
在 EEG_AI_AR (Unity 6 AR) 项目中集成云端大语言模型 (LLM) API，实现：
- **实时情绪识别与分析**：基于 EEG 数据 + 用户对话内容
- **智能情绪干预对话**：通过 AI 对话进行心理疏导和情绪调节
- **记忆恢复系统**：根据对话内容逐步解锁用户记忆碎片
- **AR 沉浸式交互**：在月球场景中通过宇航员角色进行对话

### 核心价值
1. **沉浸式叙事体验**：结合 AR 环境和 AI 对话，打造独特的月球生存故事
2. **个性化心理支持**：根据用户实时情绪状态定制对话策略
3. **记忆恢复机制**：通过对话解锁记忆碎片，推进游戏进度
4. **数据驱动**：EEG 生物信号 + 语言双重维度评估情绪状态

```csharp
        public async Task<LLMResponse> SendMessageAsync(
            List<LLMMessage> messages,
            float temperature = 0.7f,
            int maxTokens = 800)
        {
            var response = new LLMResponse();
            var startTime = Time.realtimeSinceStartup;

            try
            {
                var requestBody = new
                {
                    model = _currentModel ?? model,
                    messages = messages.Select(m => new { role = m.role, content = m.content }).ToArray(),
                    temperature = temperature,
                    max_tokens = maxTokens,
                    stream = false
                };

                string jsonBody = JsonUtility.ToJson(requestBody);
                
                using (var webRequest = new UnityWebRequest(apiUrl, "POST"))
                {
                    webRequest.uploadHandler = new UploadHandlerRaw(Encoding.UTF8.GetBytes(jsonBody));
                    webRequest.downloadHandler = new DownloadHandlerBuffer();
                    webRequest.SetRequestHeader("Content-Type", "application/json");
                    webRequest.SetRequestHeader("Authorization", $"Bearer {_currentApiKey ?? apiKey}");
                    webRequest.timeout = requestTimeoutSeconds;

                    var operation = webRequest.SendWebRequest();
                    
                    while (!operation.isDone)
                        await Task.Yield();

                    if (webRequest.result != UnityWebRequest.Result.Success)
                    {
                        throw new Exception($"API Error: {webRequest.error} - {webRequest.downloadHandler.text}");
                    }

                    string jsonResponse = webRequest.downloadHandler.text;
                    var apiResponse = JsonUtility.FromJson<OpenAIResponse>(jsonResponse);
                    
                    response.content = apiResponse.choices[0].message.content;
                    response.tokensUsed = apiResponse.usage.total_tokens;
                    response.isSuccess = true;
                    response.latencyMs = (Time.realtimeSinceStartup - startTime) * 1000;
                }
            }
            catch (Exception ex)
            {
                Debug.LogError($"[QwenService] Request failed: {ex.Message}");
                response.isSuccess = false;
                response.errorMessage = ex.Message;
            }

            return response;
        }
```

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

namespace EEGAI.Game.Memory
{
    [Serializable]
    public class MemoryFragment
    {
        public string id;
        public string title;
        public string description;
        public string content;
        public float unlockThreshold;      // 记忆恢复进度阈值
        public string[] triggerKeywords;   // 触发关键词
        public string category;            // 记忆类别（如：任务、人物、事件）
        public bool isUnlocked;
        public DateTime unlockedAt;
        
        // 视觉效果配置
        public string imagePath;           // 记忆碎片的图片
        public string audioPath;           // 记忆碎片的音频
        public float emotionalIntensity;   // 情绪强度
        
        public MemoryFragment(string id, string title, string content, float threshold)
        {
            this.id = id;
            this.title = title;
            this.content = content;
            this.unlockThreshold = threshold;
            this.isUnlocked = false;
            this.triggerKeywords = Array.Empty<string>();
        }
    }

    [Serializable]
    public class MemoryProgress
    {
        public float currentProgress;      // 0-100%
        public int unlockedFragments;
        public int totalFragments;
        public string currentPhase;        // 记忆阶段（如：觉醒、探索、真相）
        public DateTime lastUpdate;
        
        public MemoryProgress()
        {
            currentProgress = 0f;
            unlockedFragments = 0;
            totalFragments = 0;
            currentPhase = "觉醒";
            lastUpdate = DateTime.UtcNow;
        }
        
        public float GetCompletionPercentage()
        {
            return (float)unlockedFragments / totalFragments * 100f;
        }
        
        public bool IsComplete()
        {
            return currentProgress >= 100f || unlockedFragments >= totalFragments;
        }
    }

    public class MemorySystem : MonoBehaviour
    {
        [Header("Memory Configuration")]
        [SerializeField] private List<MemoryFragment> memoryFragments = new();
        [SerializeField] private float baseProgressIncrement = 5f;    // 每次对话的基础进度
        [SerializeField] private float emotionBonusMultiplier = 1.5f; // 情绪参与度的奖励倍数
        
        [Header("Game Events")]
        [SerializeField] private string completionEventName = "MemoryRecoveryComplete";
        
        private MemoryProgress progress;
        private Dictionary<string, MemoryFragment> fragmentLookup = new();
        
        public MemoryProgress Progress => progress;
        public List<MemoryFragment> UnlockedFragments => memoryFragments.FindAll(f => f.isUnlocked);
        
        private void Awake()
        {
            progress = new MemoryProgress();
            progress.totalFragments = memoryFragments.Count;
            
            foreach (var fragment in memoryFragments)
            {
                fragmentLookup[fragment.id] = fragment;
            }
        }
        
        /// <summary>
        /// 处理用户对话，更新记忆进度
        /// </summary>
        public void ProcessDialogue(string userInput, float emotionalIntensity)
        {
            // 基础进度增加
            float progressGain = baseProgressIncrement;
            
            // 情绪参与度奖励
            if (emotionalIntensity > 0.7f) // 高情绪参与
            {
                progressGain *= emotionBonusMultiplier;
            }
            
            // 检查关键词触发
            foreach (var fragment in memoryFragments)
            {
                if (!fragment.isUnlocked)
                {
                    foreach (var keyword in fragment.triggerKeywords)
                    {
                        if (userInput.Contains(keyword, StringComparison.OrdinalIgnoreCase))
                        {
                            UnlockFragment(fragment.id);
                            progressGain += 10f; // 关键词触发额外奖励
                            break;
                        }
                    }
                }
            }
            
            // 更新总进度
            progress.currentProgress = Mathf.Min(progress.currentProgress + progressGain, 100f);
            progress.lastUpdate = DateTime.UtcNow;
            
            // 检查阈值解锁
            CheckThresholdUnlocks();
            
            // 检查游戏完成
            if (progress.IsComplete())
            {
                TriggerCompletionEvent();
            }
        }
        
        /// <summary>
        /// 解锁记忆碎片
        /// </summary>
        public void UnlockFragment(string fragmentId)
        {
            if (fragmentLookup.TryGetValue(fragmentId, out var fragment) && !fragment.isUnlocked)
            {
                fragment.isUnlocked = true;
                fragment.unlockedAt = DateTime.UtcNow;
                progress.unlockedFragments++;
                
                // 触发解锁事件
                Debug.Log($"🧠 记忆碎片解锁: {fragment.title}");
                
                // 可以在这里播放解锁动画、音效等
            }
        }
        
        /// <summary>
        /// 根据进度阈值检查并解锁记忆
        /// </summary>
        private void CheckThresholdUnlocks()
        {
            foreach (var fragment in memoryFragments)
            {
                if (!fragment.isUnlocked && progress.currentProgress >= fragment.unlockThreshold)
                {
                    UnlockFragment(fragment.id);
                }
            }
        }
        
        /// <summary>
        /// 触发游戏完成事件
        /// </summary>
        private void TriggerCompletionEvent()
        {
            Debug.Log("🎉 记忆恢复完成！游戏结局触发");
            
            // 这里可以触发游戏结局的场景切换、动画等
            // 例如：
            // EventBus.Publish(completionEventName);
        }
        
        /// <summary>
        /// 获取当前阶段的记忆碎片
        /// </summary>
        public List<MemoryFragment> GetFragmentsByPhase(string phase)
        {
            return memoryFragments.FindAll(f => f.category == phase && f.isUnlocked);
        }
        
        /// <summary>
        /// 保存记忆进度
        /// </summary>
        public void SaveProgress()
        {
            // 这里实现保存逻辑，例如使用 PlayerPrefs 或 JSON
            string json = JsonUtility.ToJson(progress);
            PlayerPrefs.SetString("MemoryProgress", json);
            
            // 保存解锁状态
            var unlockedIds = new List<string>();
            foreach (var fragment in memoryFragments)
            {
                if (fragment.isUnlocked)
                {
                    unlockedIds.Add(fragment.id);
                }
            }
            PlayerPrefs.SetString("UnlockedFragments", string.Join(",", unlockedIds));
            PlayerPrefs.Save();
        }
        
        /// <summary>
        /// 加载记忆进度
        /// </summary>
        public void LoadProgress()
        {
            if (PlayerPrefs.HasKey("MemoryProgress"))
            {
                string json = PlayerPrefs.GetString("MemoryProgress");
                progress = JsonUtility.FromJson<MemoryProgress>(json);
            }
            
            if (PlayerPrefs.HasKey("UnlockedFragments"))
            {
                string unlockedStr = PlayerPrefs.GetString("UnlockedFragments");
                var unlockedIds = unlockedStr.Split(',', StringSplitOptions.RemoveEmptyEntries);
                
                foreach (var id in unlockedIds)
                {
                    UnlockFragment(id);
                }
            }
        }
    }
}
```

#### 3. 情绪分析引擎（更新版）

```csharp
using System;
using System.Collections.Generic;
using EEGAI.Game.Memory;

namespace EEGAI.AI.Emotion
{
    [Serializable]
    public enum EmotionType
    {
        Calm,           // 平静
        Happy,          // 开心
        Anxious,        // 焦虑
        Sad,            // 悲伤
        Angry,          // 愤怒
        Stressed,       // 压力大
        Confused,       // 困惑
        Neutral         // 中性
    }

    [Serializable]
    public class EmotionState
    {
        public EmotionType PrimaryEmotion;
        public float Confidence;        // 0.0 - 1.0
        public float Intensity;         // 0.0 - 1.0
        public double DurationMinutes;
        public DateTime DetectedAt;
        public Dictionary<EmotionType, float> SecondaryEmotions;
        
        // EEG 特征值（归一化后）
        public float AlphaWaveRatio;    // α 波比例（放松指标）
        public float BetaWaveRatio;     // β 波比例（警觉/焦虑指标）
        public float ThetaWaveRatio;    // θ 波比例（冥想/困倦指标
        
        public string GetDescription()
        {
            return $"{PrimaryEmotion} (强度:{Intensity:F2})";
        }
        
        public bool IsNegativeEmotion()
        {
            return PrimaryEmotion == EmotionType.Anxious ||
                   PrimaryEmotion == EmotionType.Sad ||
                   PrimaryEmotion == EmotionType.Angry ||
                   PrimaryEmotion == EmotionType.Stressed;
        }
        
        public bool RequiresIntervention()
        {
            return IsNegativeEmotion() && Intensity > 0.7f && DurationMinutes > 5.0;
        }
    }

    public class EmotionAnalyzer : MonoBehaviour
    {
        [Header("EEG Thresholds")]
        [SerializeField] private float highAnxietyBetaThreshold = 0.4f;
        [SerializeField] private float calmAlphaThreshold = 0.35f;
        
        [Header("Time Windows")]
        [SerializeField] private float analysisIntervalSeconds = 30f;
        [SerializeField] private float interventionThresholdMinutes = 5f;

        [Header("Memory Integration")]
        [SerializeField] private MemorySystem memorySystem;

        private Queue<EmotionState> emotionHistory = new Queue<EmotionState>();
        private EmotionState currentState;
        private float lastAnalysisTime;

        private void Awake()
        {
            if (memorySystem == null)
            {
                memorySystem = FindObjectOfType<MemorySystem>();
            }
        }

        /// <summary>
        /// 从原始 EEG 数据计算情绪状态
        /// </summary>
        public EmotionState AnalyzeFromEEG(EEGRawData eegData)
        {
            var state = new EmotionState
            {
                DetectedAt = DateTime.UtcNow,
                AlphaWaveRatio = CalculateBandPower(eegData, "alpha"),
                BetaWaveRatio = CalculateBandPower(eegData, "beta"),
                ThetaWaveRatio = CalculateBandPower(eegData, "theta")
            };
            
            // 基于脑电波比例推断情绪
            if (state.BetaWaveRatio > highAnxietyBetaThreshold)
            {
                state.PrimaryEmotion = EmotionType.Anxious;
                state.Confidence = Mathf.Min(state.BetaWaveRatio * 1.5f, 1.0f);
                state.Intensity = MapRange(state.BetaWaveRatio, 0.3f, 0.6f, 0.3f, 0.9f);
            }
            else if (state.AlphaWaveRatio > calmAlphaThreshold)
            {
                state.PrimaryEmotion = EmotionType.Calm;
                state.Confidence = state.AlphaWaveRatio;
                state.Intensity = 1.0f - MapRange(state.AlphaWaveRatio, 0.35f, 0.55f, 0.2f, 0.8f);
            }
            else
            {
                state.PrimaryEmotion = EmotionType.Neutral;
                state.Confidence = 0.5f;
                state.Intensity = 0.5f;
            }
            
            UpdateHistory(state);
            return state;
        }

        /// <summary>
        /// 结合文本情感增强情绪判断
        /// </summary>
        public EmotionState EnrichWithTextSentiment(EmotionState baseState, string userText, float sentimentScore)
        {
            // sentimentScore: -1.0 (非常负面) 到 1.0 (非常正面)
            if (Mathf.Abs(sentimentScore) > 0.6f)
            {
                baseState.Confidence = Mathf.Min(baseState.Confidence + 0.2f, 1.0f);
                
                if (sentimentScore < -0.6f)
                {
                    baseState.PrimaryEmotion = EmotionType.Sad;
                    baseState.Intensity = Mathf.Min(Mathf.Abs(sentimentScore), 1.0f);
                }
                else if (sentimentScore > 0.6f)
                {
                    baseState.PrimaryEmotion = EmotionType.Happy;
                    baseState.Intensity = Mathf.Min(sentimentScore, 1.0f);
                }
            }
            
            // 处理用户对话，更新记忆进度
            if (memorySystem != null)
            {
                memorySystem.ProcessDialogue(userText, baseState.Intensity);
            }
            
            return baseState;
        }

        public EmotionState GetCurrentState()
        {
            return currentState ?? new EmotionState 
            { 
                PrimaryEmotion = EmotionType.Neutral, 
                Confidence = 0.3f,
                Intensity = 0.5f
            };
        }

        public string GetEmotionTrend()
        {
            if (emotionHistory.Count < 3) return "数据不足";
            
            var recent = new List<EmotionState>(emotionHistory).TakeLast(5).ToList();
            int negativeCount = recent.Count(e => e.IsNegativeEmotion());
            
            if (negativeCount >= 4) return "持续负面，建议关注";
            if (negativeCount >= 2) return "波动较大";
            return "相对稳定";
        }

        private void UpdateHistory(EmotionState newState)
        {
            currentState = newState;
            emotionHistory.Enqueue(newState);
            
            if (emotionHistory.Count > 20) // 保留最近 20 条记录
                emotionHistory.Dequeue();
        }

        private float CalculateBandPower(EEGRawData data, string band)
        {
            // TODO: 实现 FFT 或功率谱密度计算
            // 这里返回模拟数据用于测试
            return band switch
            {
                "alpha" => UnityEngine.Random.Range(0.25f, 0.45f),
                "beta" => UnityEngine.Random.Range(0.2f, 0.5f),
                "theta" => UnityEngine.Random.Range(0.15f, 0.3f),
                _ => 0.3f
            };
        }

        private float MapRange(float value, float inMin, float inMax, float outMin, float outMax)
        {
            return Mathf.Clamp((value - inMin) / (inMax - inMin) * (outMax - outMin) + outMin, outMin, outMax);
        }
    }
```

```csharp
using UnityEngine;

        /// <summary>
        /// 处理用户输入
        /// </summary>
        public async Task ProcessUserInput(string userInput)
        {
            if (isProcessing || string.IsNullOrWhiteSpace(userInput)) return;
            
            isProcessing = true;
            
            try
            {
                // 显示用户消息
                chatUI?.AddUserMessage(userInput);
                context.AddMessage(new LLMMessage("user", userInput));
                
                // 分析情绪
                var emotionState = emotionAnalyzer.GetCurrentState();
                var enrichedState = emotionAnalyzer.EnrichWithTextSentiment(
                    emotionState, userInput, CalculateSentimentScore(userInput)
                );
                
                // 更新上下文
                UpdateContextFromMemory();
                
                // 调用 LLM
                var response = await llmService.SendMessageWithContextAsync(
                    userInput, enrichedState, context
                );
                
                if (response.isSuccess)
                {
                    // 显示 AI 回复
                    await chatUI?.AddAIMessageAsync(response.content, typingDelay);
                    
                    // 播放 TTS
                    if (enableTTS && ttsPlayer != null)
                    {
                        await ttsPlayer.PlayTextAsync(response.content);
                    }
                    
                    // 添加到上下文
                    context.AddMessage(new LLMMessage("assistant", response.content));
                }
                else
                {
                    // 显示错误消息
                    chatUI?.AddSystemMessage($"错误: {response.errorMessage}");
                }
                
                // 检查记忆解锁
                CheckMemoryUnlocks();
                
                // 保存进度
                memorySystem?.SaveProgress();
                
            } catch (Exception ex)
            {
                Debug.LogError($"[DialogueManager] Error: {ex.Message}");
                chatUI?.AddSystemMessage("系统错误，请稍后再试");
            } finally
            {
                isProcessing = false;
            }
        }

            await chatUI?.AddAIMessageAsync(welcomeMessage, typingDelay);
            
            if (enableTTS && ttsPlayer != null)
            {
                await ttsPlayer.PlayTextAsync(welcomeMessage);
            }
            
            context.AddMessage(new LLMMessage("assistant", welcomeMessage));
        }

        /// <summary>
        /// 从记忆系统更新上下文
        /// </summary>

        {
            if (memorySystem != null)
            {
                var progress = memorySystem.Progress;
                context.MemoryProgress = progress.currentProgress;
                context.UnlockedFragments = memorySystem.UnlockedFragments;
            }
        }

        /// <summary>
        /// 检查记忆解锁并显示
        /// </summary>
        private void CheckMemoryUnlocks()
        {
            var newlyUnlocked = memorySystem.UnlockedFragments.FindAll(f => 
                !context.UnlockedFragments.Exists(uf => uf.id == f.id)
            );
            
            foreach (var fragment in newlyUnlocked)
            {
                memoryUI?.ShowMemoryFragment(fragment);
            }
        }

        /// <summary>
        /// 简单的情感分析（基于关键词）
        /// </summary>
        private float CalculateSentimentScore(string text)
        {
            // 简单的关键词情感分析
            var positiveWords = new[] { "开心", "高兴", "好", "喜欢", "希望", "成功", "安全" };
            var negativeWords = new[] { "难过", "伤心", "害怕", "焦虑", "失败", "危险", "孤独" };
            
            int positiveCount = 0;
            int negativeCount = 0;
            
            foreach (var word in positiveWords)
            {
                if (text.Contains(word)) positiveCount++;
            }
            
            foreach (var word in negativeWords)
            {
                if (text.Contains(word)) negativeCount++;
            }
            
            return Mathf.Clamp((positiveCount - negativeCount) / 5f, -1f, 1f);
        }
```

```csharp
using EEGAI.Game.Memory;

namespace EEGAI.UI

    {

        [SerializeField] private GameObject memoryFragmentPrefab;
        [SerializeField] private Slider memoryProgressBar;
        [SerializeField] private Text progressText;
        
        [Header("Animation")]
        [SerializeField] private float unlockAnimationDuration = 1.5f;
        [SerializeField] private AnimationCurve unlockCurve;

        public void ShowMemoryFragment(MemoryFragment fragment)
        {
            var fragmentUI = Instantiate(memoryFragmentPrefab, memoryContainer);
            var titleText = fragmentUI.transform.Find("Title").GetComponent<Text>();
            var contentText = fragmentUI.transform.Find("Content").GetComponent<Text>();
            
            titleText.text = fragment.title;
            contentText.text = fragment.content;
            
            // 播放解锁动画
            StartCoroutine(UnlockAnimation(fragmentUI.transform));
        }

        public void UpdateProgress(float currentProgress, int unlocked, int total)
        {
            memoryProgressBar.value = currentProgress / 100f;
            progressText.text = $"记忆恢复: {unlocked}/{total} ({currentProgress:F0}%)";
        }

        private System.Collections.IEnumerator UnlockAnimation(Transform fragmentTransform)
        {
            Vector3 startScale = Vector3.zero;
            Vector3 endScale = Vector3.one;
            float elapsed = 0f;
            
            fragmentTransform.localScale = startScale;
            
            while (elapsed < unlockAnimationDuration)
            {
                float t = unlockCurve.Evaluate(elapsed / unlockAnimationDuration);
                fragmentTransform.localScale = Vector3.Lerp(startScale, endScale, t);
                elapsed += Time.deltaTime;
                yield return null;
            }
            
            fragmentTransform.localScale = endScale;
        }
    }
}
```

```csharp
public class ResilientLLMClient : MonoBehaviour
{
    [Header("Retry Strategy")]
    [SerializeField] private int maxRetries = 3;
    [SerializeField] private float baseDelaySeconds = 1f;
    [SerializeField] private float maxDelaySeconds = 10f;
    
    [Header("Fallback")]
    [SerializeField] private List<ILLMService> fallbackServices;
    
    private Queue<string> offlineCache = new(); // 离线消息队列
    
    public async Task<LLMResponse> ExecuteWithResilience(Func<Task<LLMResponse>> request)
    {
        Exception lastException = null;
        
        for (int attempt = 0; attempt <= maxRetries; attempt++)
        {
            try
            {
                if (attempt > 0)
                {
                    // 指数退避
                    float delay = Mathf.Min(
                        baseDelaySeconds * Mathf.Pow(2f, attempt - 1), 
                        maxDelaySeconds
                    );
                    await Task.Delay(TimeSpan.FromSeconds(delay));
                    
                    Debug.Log($"[ResilientClient] Retry attempt {attempt}/{maxRetries}");
                }
                
                return await request();
            }
            catch (Exception ex)
            {
                lastException = ex;
                Debug.LogWarning($"[ResilientClient] Attempt {attempt} failed: {ex.Message}");
            }
        }
        
        // 所有重试失败，尝试备用服务
        foreach (var fallback in fallbackServices)
        {
            try
            {
                Debug.Log("[ResilientClient] Trying fallback service...");
                return await request();
            }
            catch
            {
                continue;
            }
        }
        
        // 全部失败，返回预设的离线回复
        return CreateOfflineResponse();
    }
    
    private LLMResponse CreateOfflineResponse()
    {
        return new LLMResponse
        {
            content = GetRandomOfflineMessage(),
            isSuccess = true, // 对用户来说仍是"成功"的
            errorMessage = "Network unavailable, using cached response"
        };
    }
    
    private string GetRandomOfflineMessage()
    {
        var messages = new[]
        {
            "看起来网络有点不稳定呢~ 不过我还在这里陪着你。",
            "信号不太好，但我能感受到你想聊的心情。能再说说吗？",
            "月球上的通讯有时候也会延迟，别着急，我在听。"
        };
        return messages[UnityEngine.Random.Range(0, messages.Length)];
    }
}
```
